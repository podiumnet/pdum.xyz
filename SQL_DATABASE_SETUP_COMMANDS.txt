#Execute these SQL commands to get your podium database ready:
# The following command needs only to be run when the podium database doesn't
# exist yet.
CREATE DATABASE IF NOT EXISTS podium
# All commands below assume that the podium database is selected.
CREATE TABLE `podium_users` (
	`username` VARCHAR(20) NOT NULL,
	`id` BIGINT(20) UNSIGNED NOT NULL AUTO_INCREMENT,
	`password` VARCHAR(128) NOT NULL DEFAULT '' COMMENT 'SHA512 Encrypted',
	`email` VARCHAR(50) NOT NULL,
	`email_verified` BIT(1) NOT NULL DEFAULT b'0' COMMENT '1 (true) or 0 (false)',
	`password_salt` VARCHAR(11) NOT NULL DEFAULT '0',
	PRIMARY KEY (`id`),
	UNIQUE INDEX `id` (`id`),
	UNIQUE INDEX `username` (`username`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;
